// 추상화를 이해하는 것이 왜 중요하죠?

// 추상화를 하게 되면 코드의 재사용성을 높일 수 있고, 유지보수성을 높일 수 있습니다.

// function greet(name) {

// }

// 위 코드와 동일합니다.
function greet(name) {
    console.log(`안녕하세요, ${name}님!`);
}

console.log(greet("licat"));
console.log(greet("mura"));
console.log(greet("binky"));
